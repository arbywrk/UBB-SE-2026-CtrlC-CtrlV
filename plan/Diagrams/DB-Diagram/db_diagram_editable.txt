Table Users {
  id int [pk, increment]
  username varchar
  email varchar [unique]
  password_hash varchar
  created_at datetime
}

Table Events {
  id int [pk, increment]
  title varchar
  description text
  price decimal
  location varchar
  date datetime
  created_by int
  created_at datetime
}

Table Movies {
  id int [pk, increment]
  title varchar
  description text
  release_year int
  duration_minutes int
}

Table Screenings {
  id int [pk, increment]
  event_id int
  movie_id int
  screening_time datetime
}

Table Genres {
  id int [pk, increment]
  name varchar
}

Table MovieGenres {
  movie_id int
  genre_id int

  indexes {
    (movie_id, genre_id) [pk]
  }
}

Table Actors {
  id int [pk, increment]
  name varchar
}

Table MovieActors {
  movie_id int
  actor_id int

  indexes {
    (movie_id, actor_id) [pk]
  }
}

Table Directors {
  id int [pk, increment]
  name varchar
}

Table MovieDirectors {
  movie_id int
  director_id int

  indexes {
    (movie_id, director_id) [pk]
  }
}

Table EventParticipants {
  id int [pk, increment]
  user_id int
  event_id int
  joined_at datetime
}

Table EventRatings {
  id int [pk, increment]
  user_id int
  event_id int
  rating int
  created_at datetime

  indexes {
    (user_id, event_id) [unique]
  }
}

Table UserSlotMachineState {
  id int [pk, increment]
  user_id int [unique, not null]
  daily_spins_remaining int [default: 1]
  bonus_spins int [default: 0]
  login_streak int [default: 0]
  event_spin_rewards_today int [default: 0]
  updated_at datetime
}

Table UserMovieDiscount {
  id int [pk, increment]
  user_id int [not null]
  movie_id int [not null]
  discount_percentage int [not null]
  created_at datetime
}

Table AmbassadorProfile {
  id int [pk, increment]
  user_id int [unique, not null]
  referral_code varchar [unique, not null]
  reward_balance decimal [default: 0]
  created_at datetime
}

Table ReferralLog {
  id int [pk, increment]
  ambassador_id int [not null]
  friend_id int [not null]
  event_id int
  used_at datetime
}

Table TriviaQuestion {
  id int [pk, increment]
  category varchar
  question_text text
  option_a varchar
  option_b varchar
  option_c varchar
  option_d varchar
  correct_option char(1)
}

Table FavoriteEvents {
  id int [pk, increment]
  user_id int [not null]
  event_id int [not null]
  created_at datetime [default: `CURRENT_TIMESTAMP`]

  indexes {
    (user_id, event_id) [unique]
  }
}

Table UserMovieVerification {
  id int [pk, increment]
  user_id int [not null, unique]
  verified_movie_count int [default: 0]
  is_completed boolean [default: false]
  verified_movie_ids text
}
